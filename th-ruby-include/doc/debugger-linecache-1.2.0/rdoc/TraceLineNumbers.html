# File lib/tracelines19.rb, line 8
def self.lnums_for_str src
  name = "#{Time.new.to_i}_#{rand(2**31)}"
  iseq = RubyVM::InstructionSequence.compile(src, name)
  lines = {}
  iseq.disasm.each_line{|line|
    if /^\d+ (\w+)\s+.+\(\s*(\d+)\)$/ =~ line
      insn = $1
      lineno = $2.to_i
      next unless insn == 'trace'
      lines[lineno] = true
      # p [lineno, line]
    end
  }
  lines.keys
end
